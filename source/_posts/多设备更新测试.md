var arr = [2, 6, 4, 7, 11, 23, 15];
        console.log(arr.sort()) //[11, 15, 2, 23, 4, 6, 7]
        var result = arr.sort(function(a, b) {
            if (a > b) {
                return 1;
            } else if (a < b) {
                return -1;
            } else {
                return 0;
            }
        })
//或者
var result = arr.sort(function(a, b) {
            // 升序
            return a - b;
            // 降序
            // return b - a;
        })
        console.log(result) //[2, 4, 6, 7, 11, 15, 23]
